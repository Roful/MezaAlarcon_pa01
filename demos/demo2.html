36 class PairIterator:


37     def __init__(self, p):


38         self.p = p


39     def next(self):


40         if self.p == nil:


41             raise StopIteration


42         ret = self.p.fst


43         self.p = self.p.snd


44         return ret

49 class Nil:


50     def __repr__(self):


51         return "()"


52     def __iter__(self):


53         return PairIterator(self)
